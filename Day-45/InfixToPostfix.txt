a+b*c => (a+(b*c))

      => (a+(*bc))

      => (+a*bc)

      => +a*bc



(A+B) * (C-D) => (AB+) * (C-D)

              => (AB+) * (CD-)

              => AB+CD-*
              


(A-(B/C))*((A/K)-L) => (A-(BC/)) * ((A/K)-L)

                    => (A-(BC/)) * ((AK/)-L)

                    => (ABC/-) * (AK/L-)

                    => ABC/-AK/L-*
